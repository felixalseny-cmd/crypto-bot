const userSchema = new mongoose.Schema({
  userId: { type: Number, required: true, unique: true },
  username: String,
  firstName: String,
  subscription: {
    type: String,
    enum: ['none', '1month', '3months'],

// ==================== ğŸ¯ ĞšĞĞœĞĞĞ”Ğ /START ====================
bot.onText(/\/start/, async (msg) => {

  const options = {
    reply_markup: {
      inline_keyboard: [

          { text: 'ğŸ“… 1 Month - 24 USDT', callback_data: 'subscribe_1month' },
          { text: 'â­ 3 Months - 55 USDT', callback_data: 'subscribe_3months' }

// ==================== ğŸ”˜ ĞĞ‘Ğ ĞĞ‘ĞĞ¢ĞšĞ CALLBACK ĞšĞĞĞŸĞĞš ====================
bot.on('callback_query', async (callbackQuery) => {
  const message = callbackQuery.message;
  const data = callbackQuery.data;
  const chatId = message.chat.id;

  try {
    if (data.startsWith('subscribe_')) {
      const plan = data.split('_')[1];
      await sendPaymentInstructions(chatId, plan);
    } else if (data === 'my_subscription') {
      await showUserSubscription(chatId);
    } else if (data === 'how_to_pay') {
      await sendHowToPay(chatId);
    } else if (data === 'back_to_plans') {
      await bot.deleteMessage(chatId, message.message_id);
      bot.emit('message', { ...message, text: '/start' });
    }
  } catch (error) {
    console.error('Callback error:', error);
    await bot.sendMessage(chatId, 'âŒ An error occurred. Please try again.');
  }
});

// ==================== ğŸ’³ Ğ¤Ğ£ĞĞšĞ¦Ğ˜Ğ¯ ĞĞ¢ĞŸĞ ĞĞ’ĞšĞ˜ Ğ˜ĞĞ¡Ğ¢Ğ Ğ£ĞšĞ¦Ğ˜Ğ™ ĞŸĞ ĞĞŸĞ›ĞĞ¢Ğ• ====================
async function sendPaymentInstructions(chatId, plan) {
  const prices = { '1month': 24, '3months': 55 };
  const amount = prices[plan];
  const walletAddress = process.env.WALLET_ADDRESS;

  const message = `ğŸ’³ *Payment Instructions for ${plan.toUpperCase()}*\n\nğŸ“ Send exactly *${amount} USDT* (TRC20) to:\n\`${walletAddress}\`\n\nâš ï¸ *Important:*\nâ€¢ Send only USDT (TRC20)\nâ€¢ Send exact amount: *${amount} USDT*\nâ€¢ Network: *TRON (TRC20)*\nâ€¢ After payment, forward the transaction hash to this bot\n\nOnce verified, you'll get VIP access automatically!`;

  await bot.sendMessage(chatId, message, { 
    parse_mode: 'Markdown',
    reply_markup: {
      inline_keyboard: [
        [{ text: 'ğŸ”™ Back to Plans', callback_data: 'back_to_plans' }]
      ]
    }
  });

  await User.findOneAndUpdate(
    { userId: chatId },
    { 
      $set: { 
        pendingPayment: { plan, amount, timestamp: new Date() }
      }
    }
  );
}

// ==================== ğŸ“Š ĞŸĞĞšĞĞ—ĞĞ¢Ğ¬ Ğ˜ĞĞ¤ĞĞ ĞœĞĞ¦Ğ˜Ğ® Ğ ĞŸĞĞ”ĞŸĞ˜Ğ¡ĞšĞ• ====================
async function showUserSubscription(chatId) {
  const user = await User.findOne({ userId: chatId });
  
  if (!user || user.subscription === 'none') {
    await bot.sendMessage(chatId, 
      `ğŸ“Š *Your Subscription Status*\n\nâŒ No active subscription\nChoose a plan to get VIP access!`,
      { 
        parse_mode: 'Markdown',
        reply_markup: {
          inline_keyboard: [
            [{ text: 'ğŸ« View Plans', callback_data: 'back_to_plans' }]
          ]
        }
      }
    );
    return;
  }

  const remainingTime = Math.ceil((user.expiresAt - new Date()) / (1000 * 60 * 60 * 24));
  
  await bot.sendMessage(chatId,
    `ğŸ“Š *Your Subscription Status*\n\nâœ… Plan: *${user.subscription.toUpperCase()}*\nâ° Expires in: *${remainingTime} days*\nğŸ“… Renewal: *${user.expiresAt.toLocaleDateString()}*`,
    { 
      parse_mode: 'Markdown',
      reply_markup: {
        inline_keyboard: [
          [{ text: 'ğŸ”„ Renew Subscription', callback_data: 'back_to_plans' }]
        ]
      }
    }
  );
}

// ==================== ğŸ’¡ Ğ˜ĞĞ¡Ğ¢Ğ Ğ£ĞšĞ¦Ğ˜Ğ¯ ĞšĞĞš ĞĞŸĞ›ĞĞ¢Ğ˜Ğ¢Ğ¬ ====================

Send exact amount from the subscription plan\n4. Copy the *Transaction Hash (TXID)* after sending\n5. Forward the transaction hash to this bot\n\nâ³ Verification usually takes 5-15 minutes`;

  await bot.sendMessage(chatId, message, { 
    parse_mode: 'Markdown',

// ==================== ğŸ“¨ ĞĞ‘Ğ ĞĞ‘ĞĞ¢ĞšĞ Ğ¢Ğ ĞĞĞ—ĞĞšĞ¦Ğ˜Ğ™ ====================
bot.on('message', async (msg) => {
  if (msg.text && msg.text.startsWith('/')) return;
  
  const chatId = msg.chat.id;
  
  if (msg.text && msg.text.length === 64 && /^[a-fA-F0-9]+$/.test(msg.text)) {
    await bot.sendMessage(chatId, 
      `â³ Transaction received! Verifying hash: ${msg.text.substring(0, 12)}...\n\nThis may take a few minutes.`,
      { parse_mode: 'Markdown' }
    );
    
    setTimeout(async () => {
      try {
        const user = await User.findOne({ userId: chatId });
        if (user && user.pendingPayment) {
          const { plan } = user.pendingPayment;
          const expiresAt = new Date();
          expiresAt.setMonth(expiresAt.getMonth() + (plan === '1month' ? 1 : 3));
          
          await User.findOneAndUpdate(
            { userId: chatId },
            {
              subscription: plan,
              expiresAt,
              $unset: { pendingPayment: 1 },
              $push: {
                transactions: {
                  hash: msg.text,
                  amount: user.pendingPayment.amount,
                  status: 'completed',
                  timestamp: new Date()
                }
              }
            }
          );

          await bot.sendMessage(chatId,
            `âœ… *Payment Verified!*\n\nYour ${plan} VIP subscription has been activated!`,
            { parse_mode: 'Markdown' }
          );
        }
      } catch (error) {
        await bot.sendMessage(chatId, 'âŒ Error activating subscription.');
      }
    }, 10000);
  }
});
